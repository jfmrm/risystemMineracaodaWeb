function getColumnNumber(width){
    breakpointArray = window.breakpointArray || 0;
    if(breakpointArray === 0){
        console.log("Breakpoint metrics tag not included in Methode frameset. Cannot compute column widths.")
        return false;
    }else{
        threeColumn = breakpointArray [0],
        twoColumn = breakpointArray[1],
        oneColumn = breakpointArray[2];
        if(width<= oneColumn){
            return 1;
        }else if(width>oneColumn &&  width <= twoColumn){
            return 2;
        }else if (width>twoColumn &&  width <= threeColumn) {
            return 3;
        }else if (width>threeColumn){
            return 4;
        }

    }

}
